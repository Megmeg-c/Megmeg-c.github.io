const scoreBoard = document.getElementById('scoreBoard'); const gameWidth = 600; const gameHeight = 400; const basketWidth = 60; let basketX = 270; let score = 0; let foodInterval; let gameOverFlag = false; // 5種小吃圖片 const foods = [ 'images/bubble_tea.png', 'images/stinky_tofu.png', 'images/fried_chicken.png', 'images/shaved_ice.png', 'images/spring_roll.png' ]; // 移動籃子 document.addEventListener('keydown', (e) => { if (e.key === 'ArrowLeft') { basketX = Math.max(0, basketX - 20); } else if (e.key === 'ArrowRight') { basketX = Math.min(gameWidth - basketWidth, basketX + 20); } basket.style.left = basketX + 'px'; }); // 開始遊戲 startBtn.addEventListener('click', () => { startBtn.disabled = true; score = 0; scoreBoard.innerText = '分數: ' + score; gameOverFlag = false; startFallingFood(); }); // 生成食物並掉落 function startFallingFood() { foodInterval = setInterval(() => { if (gameOverFlag) return; const food = document.createElement('div'); food.classList.add('food'); const foodImg = foods[Math.floor(Math.random() * foods.length)]; food.style.backgroundImage = `url(${foodImg})`; const foodX = Math.random() * (gameWidth - 50); let foodY = 0; food.style.left = foodX + 'px'; food.style.top = foodY + 'px'; gameArea.appendChild(food); const fallInterval = setInterval(() => { if (gameOverFlag) { clearInterval(fallInterval); return; } foodY += 5; food.style.top = foodY + 'px'; // 撞到籃子 if ( foodY + 50 >= gameHeight - 60 && foodX + 50 >= basketX && foodX <= basketX + basketWidth ) { score++; scoreBoard.innerText = '分數: ' + score; gameArea.removeChild(food); clearInterval(fallInterval); } // 掉到地板 else if (foodY > gameHeight) { gameOver(); clearInterval(fallInterval); } }, 50); }, 1000); } // 遊戲結束 function gameOver() { gameOverFlag = true; alert('遊戲結束！你的分數: ' + score); startBtn.disabled = false; document.querySelectorAll('.food').forEach(f => f.remove()); }
